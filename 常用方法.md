## 前端

~~~js
//时间校验
//html:onvalidation="onAfterTodayDate"
var onAfterTodayDate = function(e) {
	if (e.isValid) {
		var currentDate = new Date(e.value);
		var now = new Date();
		if (currentDate <= now) {
			e.isValid = false;
			e.errorText = "必须设置晚于当前时间的时间";
		}
	}
}
//精确时间yyyy-MM-dd HH:mm:ss
~~~
